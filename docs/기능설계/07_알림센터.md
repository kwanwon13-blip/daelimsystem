## server.js — notify() 함수 + API
```js
function notify(targetUserId, type, message, link = '') {
  const notifs = db['알림'].load();
  notifs.notifications.push({
    id: `notif_${Date.now()}`,
    대상: targetUserId, 유형: type,
    메시지: message, 링크: link,
    읽음: false, 생성시간: new Date().toISOString()
  });
  // 사용자당 최근 100개만 유지
  const mine = notifs.notifications.filter(n => n.대상 === targetUserId);
  if (mine.length > 100) {
    const cutoff = mine[mine.length - 100].id;
    notifs.notifications = notifs.notifications.filter(n => n.대상 !== targetUserId || n.id >= cutoff);
  }
  db['알림'].save(notifs);
}

// 내 알림 목록
app.get('/api/notifications', requireAuth, (req, res) => {
  const notifs = db['알림'].load();
  const mine = notifs.notifications.filter(n => n.대상 === req.session.userId).reverse().slice(0, 50);
  res.json({ unreadCount: mine.filter(n => !n.읽음).length, notifications: mine });
});

// 읽음 처리
app.put('/api/notifications/:id/read', requireAuth, (req, res) => {
  const notifs = db['알림'].load();
  const n = notifs.notifications.find(n => n.id === req.params.id);
  if (n) { n.읽음 = true; db['알림'].save(notifs); }
  res.json({ ok: true });
});
```

## notify() 사용 예시
```js
// 결재 요청 시 → admin들에게 알림
db['조직관리'].load().users.filter(u => u.role === 'admin').forEach(admin => {
  notify(admin.username, 'approval', `${req.session.name}님이 결재를 요청했습니다`, 'approval');
});

// 견적 승인 시 → 요청자에게 알림
notify(quote.createdBy, 'quote', `견적서 "${quote.quoteName}"이 승인되었습니다`, 'history');
```

## 프론트 — index.html 상단 바 (네비게이션 영역)
```html
<!-- 알림 벨 -->
<div class="relative cursor-pointer" @click="showNotifPanel = !showNotifPanel">
  <span class="material-symbols-outlined">notifications</span>
  <span x-show="unreadCount > 0"
    class="absolute -top-1 -right-1 bg-red-500 text-white text-xs rounded-full w-4 h-4 flex items-center justify-center"
    x-text="unreadCount"></span>
</div>

<!-- 알림 드롭다운 -->
<div x-show="showNotifPanel" class="absolute right-0 top-10 w-80 bg-white shadow-xl rounded-lg z-50 border">
  <template x-for="n in notifications" :key="n.id">
    <div :class="n.읽음 ? 'opacity-60' : 'bg-blue-50'"
         class="p-3 border-b cursor-pointer hover:bg-gray-50"
         @click="readNotif(n); activeTab = n.링크">
      <div class="text-sm" x-text="n.메시지"></div>
      <div class="text-xs text-gray-400 mt-1" x-text="new Date(n.생성시간).toLocaleString('ko')"></div>
    </div>
  </template>
</div>
```

## index.html function app() 추가 state/method
```js
notifications: [], unreadCount: 0, showNotifPanel: false,
async loadNotifications() {
  const res = await fetch('/api/notifications');
  const d = await res.json();
  this.notifications = d.notifications;
  this.unreadCount = d.unreadCount;
},
async readNotif(n) {
  await fetch(`/api/notifications/${n.id}/read`, { method: 'PUT' });
  n.읽음 = true;
  this.unreadCount = Math.max(0, this.unreadCount - 1);
},
// init() + setInterval(30초)에서 loadNotifications() 호출
```
