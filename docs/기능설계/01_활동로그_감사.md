| 파일 | 작업 |
|------|------|
| `db.js` | jsonStores에 `'활동로그': { logs: [] }` 1줄 추가 |

## db.js 수정 (1줄)
```js
const jsonStores = {
  '조직관리': { users: [], departments: [] },
  '활동로그': { logs: [] },  // ← 추가
};
```

## server.js — auditLog 함수
```js
function auditLog(userId, action, target, detail = {}) {
  const logs = db['활동로그'].load();
  logs.logs.push({
    id: `log_${Date.now()}`,
    시간: new Date().toISOString(),
    사용자: userId,
    행동: action,   // 예: "품목 수정", "견적 삭제"
    대상: target,   // 예: "현수막 (BN)"
    상세: detail,   // 예: { 변경전: "1000", 변경후: "1500" }
  });
  if (logs.logs.length > 10000) logs.logs = logs.logs.slice(-10000);
  db['활동로그'].save(logs);
}
```

## server.js — API에 삽입 예시
```js
// 품목 수정
app.put('/api/categories/:id', requireAuth, (req, res) => {
  const before = db.sql.categories.getById(req.params.id);
  const result = db.sql.categories.update(req.params.id, req.body);
  auditLog(req.session.userId, '품목 수정', before.name, { 변경전: before, 변경후: result });
  res.json(result);
});
// 견적 삭제, 로그인 등 동일 패턴
```

## server.js — 조회 API
```js
app.get('/api/audit-logs', requireAuth, requireAdmin, (req, res) => {
  const logs = db['활동로그'].load();
  const { page = 1, limit = 50 } = req.query;
  const sorted = [...logs.logs].reverse();
  res.json({ total: sorted.length, logs: sorted.slice((page-1)*limit, page*limit) });
});
```
